### Functions and Fat Arrow Syntax

```js
var x = function() {
  console.log('hello');
  return 'Passed from x';
};

function y() {
  console.log('world');
}

var z = () => {
  console.log('yup');
};

var a = () => {
  return 'no';
};


console.log(x); //
x(); //
console.log(x()); //
console.log(y()); //
console.log(z()); //
console.log(a()); //
```
